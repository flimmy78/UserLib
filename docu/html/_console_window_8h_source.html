46  void SetScrollBarInfo(sdword max, sword page)
47  {
48  Minimum = 0;
49  if ( page < 20 )
50  {
51  if ( max < page )
52  Maximum = page;
53  else if ( max < (page << 1) )
54  Maximum = (page << 1);
55  else if ( max < 600 )
56  Maximum = max + ((max / 30) * page);
57  else
58  Maximum = max + ((max / 60) * page);
59  }
60  else if ( page < 40 )
61  {
62  if ( max < page )
63  Maximum = page;
64  else if ( max < (page << 1) )
65  Maximum = (page << 1);
66  else if ( max < 600 )
67  Maximum = max + ((max / 60) * page);
68  else
69  Maximum = max + ((max / 120) * page);
70  }
71  else if ( page < 60 )
72  {
73  if ( max < page )
74  Maximum = page;
75  else if ( max < (page << 1) )
76  Maximum = (page << 1);
77  else if ( max < 600 )
78  Maximum = max + ((max / 90) * page);
79  else
80  Maximum = max + ((max / 150) * page);
81  }
82  else
83  {
84  if ( max < page )
85  Maximum = page;
86  else if ( max < (page << 1) )
87  Maximum = (page << 1);
88  else if ( max < 600 )
89  Maximum = max + ((max / 150) * page);
90  else
91  Maximum = max + ((max / 240) * page);
92  }
93  Current = 0;
94  ThumbSize = Cast(sword, (Cast(sdword,page) * Cast(sdword, page)) / Maximum);
95  if ( ThumbSize < 1 )
96  ThumbSize = 1;
97  LineStep = page / 5;
98  PageStep = page;
99  }
100  };
